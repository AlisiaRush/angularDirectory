Parent Component

HTML FILE:
<input type="text" #childValue >
<button class="btn btn-primary" (click)="getChildValue(childValue)"> Get Child Value </button>
<app-child-component [childValue]="value"></app-child-component>

TYPESCRIPT FILE:
public value: any = 'Child Value';

public getChildValue(inputValue: HTMLInputElement) {
this.value = inputValue.value;
console.log(this.value);
}

HTML FILE:
<p>The child value is: { { value } } </p>
TYPESCRIPT FILE:
@Input() childValue: string = 'Child Value';
